import tkinter as tk from tkinter import messagebox def click_boton(valor): pantalla.insert(tk.END, valor) def borrar(): pantalla.delete(0, tk.END) def calcular(): try: resultado = eval(pantalla.get()) borrar() pantalla.insert(tk.END, str(resultado)) except ZeroDivisionError: messagebox.showerror("Error", "No se puede dividir entre cero") borrar() except Exception: messagebox.showerror("Error", "Expresión inválida") borrar() ventana = tk.Tk() ventana.title("Calculadora") ventana.geometry("350x450") ventana.configure(bg="#202020") ventana.resizable(False, False) pantalla = tk.Entry(ventana, font=("Arial", 24), bd=10, insertwidth=4, width=14, borderwidth=0, justify="right") pantalla.grid(row=0, column=0, columnspan=4, padx=15, pady=20) estilo_botones = { "font": ("Arial", 16, "bold"), "fg": "#ffffff", "bg": "#333333", "activebackground": "#444444", "activeforeground": "#ffffff", "borderwidth": 0, "height": 2, "width": 5 } estilo_operadores = estilo_botones.copy() estilo_operadores.update({"bg": "#ff9500", "activebackground": "#ffaa66"}) estilo_borrar = estilo_botones.copy() estilo_borrar.update({"bg": "#a5a5a5", "fg": "#000000", "activebackground": "#d4d4d4"}) botones = [ ('C', 1, 0, estilo_borrar), ('/', 1, 3, estilo_operadores), ('7', 2, 0, estilo_botones), ('8', 2, 1, estilo_botones), ('9', 2, 2, estilo_botones), ('*', 2, 3, estilo_operadores), ('4', 3, 0, estilo_botones), ('5', 3, 1, estilo_botones), ('6', 3, 2, estilo_botones), ('-', 3, 3, estilo_operadores), ('1', 4, 0, estilo_botones), ('2', 4, 1, estilo_botones), ('3', 4, 2, estilo_botones), ('+', 4, 3, estilo_operadores), ('0', 5, 0, estilo_botones), ('.', 5, 1, estilo_botones), ('=', 5, 2, estilo_operadores) ] for (texto, fila, col, estilo) in botones: if texto == '=': btn = tk.Button(ventana, text=texto, command=calcular, **estilo) btn.grid(row=fila, column=col, columnspan=2, sticky="nsew", padx=5, pady=5) elif texto == 'C': btn = tk.Button(ventana, text=texto, command=borrar, **estilo) btn.grid(row=fila, column=col, columnspan=3, sticky="nsew", padx=5, pady=5) else: btn = tk.Button(ventana, text=texto, command=lambda t=texto: click_boton(t), **estilo) btn.grid(row=fila, column=col, padx=5, pady=5) for i in range(4): ventana.grid_columnconfigure(i, weight=1) ventana.mainloop()
